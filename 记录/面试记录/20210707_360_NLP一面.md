### coding
二维矩阵的二分查找
```python
#!/usr/bin/env python
'''
1 2 4
3 4 5
7 8 9
'''

def main(matrix, target):
    m, n = len(matrix), len(matrix[0])
    i, j = m-1, 0
    while 0 <= i < m and 0 <= j < n:
        cand = matrix[i][j]
        if cand == target: return i, j
        if cand < target:
            j += 1
        else:
            i -= 1
    return -1

if __name__ == '__main__':
    matrix = [
        [1, 2, 4],
        [3, 4, 5],
        [7, 8, 9]]

    print(main(matrix, 3))
```
